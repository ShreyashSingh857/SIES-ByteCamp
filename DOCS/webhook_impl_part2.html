import { getSession } from '../config/neo4j.js'; export async function patchNeo4jGraph(scanId, partialGraph, { removedFiles = [] }) { const session = getSession(); try { // 1. Delete nodes for removed files and orphaned children for (const filePath of removedFiles) { await session.run(` MATCH (f:File { path: $path, scanId: $scanId }) OPTIONAL MATCH (f)<-[:CONTAINS]-(fn:Function) DETACH DELETE f, fn `, { path: filePath, scanId }); } // 2. Upsert file nodes (MERGE = create if missing, update if exists) const fileNodes = (partialGraph.nodes || []).filter(n => n.type === 'FILE'); for (const file of fileNodes) { await session.run(` MERGE (f:File { id: $id }) SET f += $props SET f.scanId = $scanId SET f.updatedAt = datetime() `, { id: file.id, props: { path: file.name, language: file.language || 'unknown' }, scanId }); } // 3. Upsert function nodes const fnNodes = (partialGraph.nodes || []).filter(n => n.type === 'FUNCTION'); for (const fn of fnNodes) { await session.run(` MERGE (f:Function { id: $id }) SET f += $props SET f.scanId = $scanId SET f.updatedAt = datetime() `, { id: fn.id, props: { name: fn.name, qualifiedName: fn.qualifiedName || fn.name }, scanId }); } // 4. Drop stale edges from all changed file paths, then re-add current ones const changedPaths = fileNodes.map(n => n.name); await session.run(` MATCH (f:File { scanId: $scanId })-[r:IMPORTS|CALLS|DEPENDS_ON]->() WHERE f.path IN $changedPaths DELETE r `, { scanId, changedPaths }); for (const edge of (partialGraph.edges || [])) { const relType = ['IMPORTS','CALLS','DEPENDS_ON'].includes(edge.type) ? edge.type : 'DEPENDS_ON'; await session.run(` MATCH (a { id: $from, scanId: $scanId }) MATCH (b { id: $to, scanId: $scanId }) MERGE (a)-[r:${relType}]->(b) SET r.updatedAt = datetime() `, { from: edge.from || edge.source, to: edge.to || edge.target, scanId }); } return { patched: fileNodes.length + fnNodes.length, removed: removedFiles.length, edges: partialGraph.edges?.length || 0, }; } finally { await session.close(); } } export async function getLastSyncInfo(scanId) { const session = getSession(); try { const result = await session.run(` MATCH (s:Scan { id: $scanId }) RETURN s.lastWebhookSync AS lastSync, s.commitSha AS commit `, { scanId }); return result.records[0]?.toObject() || null; } finally { await session.close(); } }

const clients = new Map(); // repoId -> Set<res> export function addClient(repoId, res) { res.writeHead(200, { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache', 'Connection': 'keep-alive', 'X-Accel-Buffering': 'no', // disables nginx buffering }); res.write(':ok\n\n'); // initial heartbeat if (!clients.has(repoId)) clients.set(repoId, new Set()); clients.get(repoId).add(res); // Heartbeat every 25s to prevent proxy timeouts const hb = setInterval(() => res.write(':ping\n\n'), 25000); res.on('close', () => { clearInterval(hb); clients.get(repoId)?.delete(res); }); } export function broadcast(repoId, payload) { const group = clients.get(repoId); if (!group?.size) return; const data = `data: ${JSON.stringify(payload)}\n\n`; group.forEach(res => { try { res.write(data); } catch { group.delete(res); } }); } export function clientCount(repoId) { return clients.get(repoId)?.size || 0; }
