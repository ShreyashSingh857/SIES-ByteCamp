import crypto from 'crypto'; import { getWebhookSubscription } from '../db/webhook.queries.js'; import { webhookQueue } from '../queues/webhook.queue.js'; export const handleGithubWebhook = async (req, res, next) => { try { const event = req.headers['x-github-event']; const sig = req.headers['x-hub-signature-256']; const body = req.body; if (!sig) return res.status(401).json({ error: 'Missing signature' }); const repoFullName = body.repository?.full_name; const sub = await getWebhookSubscription(repoFullName); if (!sub) return res.status(404).json({ error: 'Unknown repo' }); // Verify signature — MUST use raw body bytes, not re-serialised JSON const expected = 'sha256=' + crypto .createHmac('sha256', sub.secret) .update(req.rawBody) // see middleware note below .digest('hex'); if (!crypto.timingSafeEqual( Buffer.from(sig.padEnd(71)), Buffer.from(expected.padEnd(71)) )) { return res.status(401).json({ error: 'Signature mismatch' }); } if (event === 'push') { const changedFiles = body.commits .flatMap(c => [...(c.added||[]), ...(c.modified||[]), ...(c.removed||[])]) const removed = body.commits.flatMap(c => c.removed || []); await webhookQueue.add('repo-update', { repoId: sub.repoId, scanId: sub.scanId, repoFullName, localPath: sub.localPath, before: body.before, after: body.after, ref: body.ref, changedFiles: [...new Set(changedFiles)], removedFiles: [...new Set(removed)], }, { attempts: 3, backoff: { type: 'exponential', delay: 5000 } }); } res.status(200).json({ ok: true }); } catch (err) { next(err); } };
