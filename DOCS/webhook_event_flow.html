1
Developer pushes a commit to GitHub
GitHub fires a push event to your registered webhook URL. The payload includes the before/after commit SHAs and a list of all added, modified, and removed files.
POST https://yourapp.com/api/webhook/github X-GitHub-Event: push X-Hub-Signature-256: sha256=abc123... { "ref": "refs/heads/main", "before": "a1b2c3d", "after": "e4f5g6h", "repository": { "full_name": "org/repo" }, "commits": [{ "added": ["src/newFeature.js"], "modified": ["src/api/users.js"], "removed": ["src/legacy/old.js"] }] }
2
Webhook receiver validates and enqueues
security Verify the HMAC-SHA256 signature using your stored webhook secret before touching any data. Then extract changed files and push a job into the queue. Respond 200 immediately — GitHub retries if you take >10 seconds.
// Backend/src/controllers/webhook.controller.js import crypto from 'crypto'; import { webhookQueue } from '../queues/webhook.queue.js'; export const handleGithubWebhook = async (req, res, next) => { const sig = req.headers['x-hub-signature-256']; const secret = await getWebhookSecret(req.body.repository.full_name); const expected = 'sha256=' + crypto .createHmac('sha256', secret) .update(JSON.stringify(req.body)) .digest('hex'); if (!crypto.timingSafeEqual(Buffer.from(sig), Buffer.from(expected))) { return res.status(401).json({ error: 'Invalid signature' }); } const changedFiles = req.body.commits .flatMap(c => [...c.added, ...c.modified, ...c.removed]); await webhookQueue.add('repo-update', { repoFullName: req.body.repository.full_name, before: req.body.before, after: req.body.after, ref: req.body.ref, changedFiles: [...new Set(changedFiles)], }); res.status(200).json({ received: true }); // reply fast };
3
Diff worker picks up the job
async The worker runs in a separate process. It does a git fetch + git diff to confirm exactly which files changed, then calls the AI-Engine with only those files rather than the whole repo — making rescans 10-100x faster.
// Backend/src/workers/webhook.worker.js import { Worker } from 'bullmq'; import simpleGit from 'simple-git'; import { runPartialScan } from '../services/partialScan.service.js'; import { patchNeo4jGraph } from '../services/neo4j-patch.service.js'; import { broadcast } from '../services/sse.service.js'; new Worker('webhook-jobs', async (job) => { const { repoFullName, before, after, changedFiles } = job.data; // 1. Pull latest changes into the already-cloned repo const repoRecord = await getRepoRecord(repoFullName); const git = simpleGit(repoRecord.localPath); await git.fetch('origin'); await git.checkout(after); // 2. Run AI-Engine only on changed files const partialGraph = await runPartialScan(repoRecord.localPath, changedFiles); // 3. Patch Neo4j — merge new nodes, delete removed ones const patch = await patchNeo4jGraph(repoRecord.scanId, partialGraph, { removedFiles: changedFiles.filter(f => !existsAfterCommit(f)), }); // 4. Push the diff to all connected browser clients broadcast(repoRecord.repoId, { type: 'GRAPH_PATCH', patch }); }, { connection: redisConnection });
4
Neo4j graph gets surgically patched
database Instead of dropping and re-seeding the entire graph, use Cypher MERGE for upserts and MATCH...DELETE for removed nodes. This keeps existing relationships intact and only modifies what changed.
// Backend/src/services/neo4j-patch.service.js export async function patchNeo4jGraph(scanId, partialGraph, { removedFiles }) { const session = getSession(); try { // Upsert modified/added file nodes for (const file of partialGraph.nodes.filter(n => n.type === 'FILE')) { await session.run(` MERGE (f:File { id: $id }) SET f += $props, f.scanId = $scanId, f.updatedAt = datetime() `, { id: file.id, props: file, scanId }); } // Delete nodes for removed files + their orphaned children for (const filePath of removedFiles) { await session.run(` MATCH (f:File { path: $path, scanId: $scanId }) OPTIONAL MATCH (f)<-[:CONTAINS]-(fn:Function) DETACH DELETE f, fn `, { path: filePath, scanId }); } // Upsert new edges, remove stale ones from changed files await session.run(` MATCH (:File { scanId: $scanId })-[r:IMPORTS|CALLS]->() WHERE r.sourceFile IN $changedFiles DELETE r `, { scanId, changedFiles: partialGraph.changedFilePaths }); for (const edge of partialGraph.edges) { await session.run(` MATCH (a { id: $from, scanId: $scanId }) MATCH (b { id: $to, scanId: $scanId }) MERGE (a)-[r:${edge.type}]->(b) SET r.updatedAt = datetime() `, { from: edge.from, to: edge.to, scanId }); } return { patched: partialGraph.nodes.length, removed: removedFiles.length }; } finally { await session.close(); } }
5
SSE pushes the graph patch to the browser
realtime Server-Sent Events keep a persistent HTTP connection open from each browser tab. When Neo4j is patched, the worker calls broadcast() and every connected client gets the diff instantly — no polling.
// Backend/src/services/sse.service.js const clients = new Map(); // repoId -> Set of response objects export function addClient(repoId, res) { res.writeHead(200, { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache', 'Connection': 'keep-alive', }); res.write('data: {"type":"connected"}\n\n'); if (!clients.has(repoId)) clients.set(repoId, new Set()); clients.get(repoId).add(res); res.on('close', () => clients.get(repoId)?.delete(res)); } export function broadcast(repoId, payload) { const group = clients.get(repoId); if (!group) return; const data = `data: ${JSON.stringify(payload)}\n\n`; group.forEach(res => res.write(data)); }
6
React applies the patch — graph updates live
frontend A single useEffect in GraphView.jsx opens an EventSource connection. When a GRAPH_PATCH event arrives, it dispatches to Redux which updates Cytoscape via the existing data flow — no page reload.
// Frontend/src/pages/GraphView.jsx (add this useEffect) useEffect(() => { if (!currentRepoId) return; const es = new EventSource(`/api/events/${currentRepoId}`); es.onmessage = (event) => { const msg = JSON.parse(event.data); if (msg.type === 'GRAPH_PATCH') { dispatch(applyGraphPatch(msg.patch)); // new Redux action } if (msg.type === 'SCAN_COMPLETE') { dispatch(refetchGraph(currentRepoId)); // trigger RTK Query refetch } }; es.onerror = () => es.close(); return () => es.close(); }, [currentRepoId, dispatch]);
